🔥 Firebase: Link to Another House
// MOVE: Remove from current, add to new
const userRef = doc(db, "users", phone);
const userData = (await getDoc(userRef)).data();
const oldHouseId = userData.house_id;

// Remove from old house's member list
await updateDoc(doc(db,"houses",oldHouseId), {
  member_phones: arrayRemove(phone)
});

// Add to new house
await updateDoc(doc(db,"houses",newHouseId), {
  member_phones: arrayUnion(phone)
});

// Update user's primary house_id
await updateDoc(userRef, {
  house_id: newHouseId,
  prev_house_id: oldHouseId
});

// ADD: Keep in both (dual membership)
await updateDoc(userRef, {
  secondary_house_ids: arrayUnion(newHouseId)
});
